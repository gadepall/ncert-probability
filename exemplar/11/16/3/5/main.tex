\begin{align}
	p_{X}(k)= \begin{cases} 
      2p, & if\ k=2m-1 \\
      p, & if\ k=2m 
   \end{cases}
\end{align}
Since $1\leq X \leq6$,
\begin{align}
	\sum_{i=1}^{6}{\pr{X=i}} &= 1\\
\implies 6p+3p &= 1\\
\implies p &= \frac{1}{9}
\end{align}
The CDF
\begin{align}
F_X (k)= \begin{cases} 
      \frac{3k+1}{18}, & if\ k=2m-1 \\
      \frac{k}{6}, & if\ k=2m 
   \end{cases}
\end{align}
Thus, 
\begin{align}
  \pr{G} &= \pr{X>3}
  =F_X (6) - F_X (3)\\
  &=1 - \frac{3(3)+1}{18}
  = \frac{4}{9}
  \end{align}
