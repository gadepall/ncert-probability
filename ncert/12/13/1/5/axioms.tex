\subsection*{Solution}

\begin{enumerate}
	\item Since
\begin{align}
	\pr{AB} &= \pr{A} + \pr{B} -\pr{A+B},
	\label{eq:1}
	\\
	\pr{AB} &= \frac{6}{11} + \frac{5}{11} - \frac{7}{11}
 = \frac{4}{11}
\end{align}
	\item Since
\begin{align}
	\pr{A \mid B} &= \frac{\pr{AB}}{\pr{B}},
	\label{eq:2}
	\\
	\pr{A \mid B} &= \frac{\frac{4}{11}}{\frac{5}{11}}
 = \frac{4}{5}
\end{align}
	\item Since
\begin{align}
	\pr{B \mid A}&=\frac{\pr{BA}}{\pr{A}},
	\label{eq:3}
	\\
	\pr{B \mid A} &= \frac{\frac{4}{11}}{\frac{6}{11}}
	 = \frac{2}{3}
\end{align}
\end{enumerate}
