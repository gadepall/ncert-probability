\begin{enumerate}
\item  
\begin{align}
	\pr {A+B} &=  \pr A + \pr B - \pr{AB} \\&=  0.88
\end{align}
\item By De Morgan's Law,
\begin{align}
	A^\prime B^\prime &= {(A+B)}^\prime\\
	\implies \pr {A^\prime B^\prime} &= \pr{A+B}^\prime \\&= 1 - \pr {A +B}\\
	&= 0.12
\end{align}
\item 
	From 
\eqref{eq:axiom_sum_A},
\begin{align}
\pr {A } 
	 &= \pr{AB}+\pr{AB^\prime} \\
	 \implies \pr{AB^\prime} 
	&= 0.19
\end{align}
\item Similarly, 
\begin{align}
	\pr {B A^\prime} &= \pr {B} - \pr{AB} 
= 0.34.
\end{align}
\end{enumerate}
