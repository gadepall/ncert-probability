\begin{align}
    \pr{B|A} &= \frac{\pr{AB}}{\pr{A}}\\
    &= \frac{\pr{A} + \pr{B} - \pr{A+B}}{\pr{A}}
    \\
    &= \frac{\pr{A} + 1-\pr{B^\prime} - \pr{A+B}}{\pr{A}}
    \\
    &= 1-\frac{ \pr{B^\prime} }{\pr{A}}
+\frac{  1- \pr{A+B}}{\pr{A}}
	\label{eq:12/13/3/102}
\end{align}
From 
	\eqref{eq:axiom-pos}
\begin{align}
{  1- \pr{A+B}} \ge 0
\end{align}
Using this in 
	\eqref{eq:12/13/3/102}
	results in 
	\eqref{eq:12/13/3/102/final}.
