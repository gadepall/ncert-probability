\begin{itemize}
	\item[a)]	If $A \subset B $ and $\pr{B}\neq 0 $ then 
\begin{align}
	AB &=A\\
	\text{or, } P(A|B) &= \frac{\pr{A B}}{\pr{B}}=\frac{\pr{A}}{\pr{B}}&\label{eq:12/13/3/143.0.1}
	%\implies \pr{A} &< \pr{B}\\
\end{align}
	\item[c)]
Since
\begin{align}
	\pr{B} &\leq 1,
	\\
	1 &\leq \frac{1}{\pr{B}}
	\\
\implies	\pr{A} &\leq \frac{\pr{A}}{\pr{B}}
\\
	&= \pr{A|B}
\end{align}
from  \eqref{eq:12/13/3/143.0.1}.
\end{itemize}
