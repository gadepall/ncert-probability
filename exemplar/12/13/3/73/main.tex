\begin{align}
    \pr{EF} = \pr{E}\pr{F}\\
	\therefore \pr{F} = \frac{\pr{E+F}-\pr{E}}{{1 - \pr{E}}} 
 = \frac{2}{7}
\end{align}
using 
\eqref{eq:axiom_sum_AB}
and simplifying.
From 
\eqref{eq:axiom-cond},
\begin{align}
    \pr{E|F} = \pr{E},\
    \pr{F|E} = \pr{F}\\
\implies \pr{E|F} - \pr{F|E} = \pr{E} - \pr{F}
    = \frac{1}{70}
\end{align}
