\begin{enumerate}[label=\thesubsection.\arabic*,ref=\thesubsection.\theenumi]
\item Let
\begin{align}
	X \sim \bnm{n}{p}
\end{align}
The mean and variance are then given by 
\begin{align}
    \mu = np,\,
    \sigma^2 = npq.
\end{align}
\item For large $n$, 
\begin{align}
	X \system*{d} Y \sim \gauss{\mu}{\sigma}
\end{align}
which implies that $X$ converges in distribution to the Gaussian.  The Gaussian distribution has the probability density function (PDF)
\begin{align}
	p_Y(x) = \frac{1}{\sqrt{2\pi} \sigma}e^{-\frac{\brak{x-\mu}^2}{2\sigma^2}}.
\end{align}
