\begin{enumerate}
    \item 
	    \begin{align}   
   \pr{B} = \frac{x}{12},
\end{align}  
  \item After adding 6 more black balls, the total number of balls is 18 and 

the number of black balls is $x+6$.  Hence, 
\begin{align}  
  \pr{B} &= \frac{x+6}{18} 
\end{align}  
From the given information,
\begin{align}
  \frac{x+6}{18} &=  \frac{2x}{12} \\
 \implies  x &= 3
\end{align}
\end{enumerate}
