Let 
\begin{align}
X= \begin{cases} 
      1 & \text{if drawn ball is red} \\
      0 & \text{otherwise.} 
   \end{cases}
\end{align}
\begin{enumerate}[label=(\roman*)]
    \item Probability that the drawn ball is red
    \begin{align}
    \pr{X=1}
    = \frac{3}{8}\\
    \end{align}
    \item Probability that the drawn ball is not red
    \begin{align}
     \pr{X=0}= 1-\frac{3}{8} = \frac{5}{8}
    \end{align}
\end{enumerate}
