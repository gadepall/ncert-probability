\begin{align}
\sum_{i=1}^{n} p_X(i) = 1
\\
\implies 50k &= 1\\
\text{or, }k &= 0.02
\end{align}
\item 
\begin{align}
E\brak{X} 
= 260k\\
= 5.2
\end{align}
\item 
\begin{align}
	\sigma_X^2 &= 2.92
\\
\implies \sigma_X &= 1.7
\end{align}
\end{enumerate}
