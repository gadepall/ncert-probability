\noindent\textbf{12.13.1.8 Question: Compute $\pr{E|F}$ where E and F are:
A die is thrown three times,
E : 4 appears on the third toss,
F : 6 and 5 appears respectively
on first two tosses. }\\
\bigskip\\
\noindent\textbf{Answer: }($\frac{1}{6}$)\\\\
\noindent\textbf{Solution: }
\\F is the event that 6 and 5 occur respectively in first two tosses.
\begin{align}
    \pr{F}=\frac{1}{6}\times\frac{1}{6}\\
    \pr{F}=\frac{1}{36}
\end{align}
The event EF consists of the only outcome {6,5,4} occuring in first three tosses.
\begin{align}
    \pr{EF}=\frac{1}{6}\times\frac{1}{6}\times\frac{1}{6}\\
    \pr{EF}=\frac{1}{216}
\end{align}
\begin{table}[htbp]
  \label{table:1}
  \input{tables/table1.tex}
  \noindent\caption{Final probabilities of the events.}
  \end{table}
We have,\\
\begin{align}
\pr{E|F}=\frac{\pr{EF}}{\pr{F}}\\
\pr{E|F}=\frac{\frac{1}{216}}{\frac{1}{36}}\\
\pr{E|F}=\frac{1}{6}
\end{align}
\bigskip
