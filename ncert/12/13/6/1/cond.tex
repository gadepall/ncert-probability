		We use 
\begin{align}
\pr{B \mid A}=\frac{\pr{BA}}{\pr{A}} \label{eq:ncert/12/13/6/1/1}
\end{align}
\begin{enumerate}
\item In this case,
\begin{align}
BA &= A \implies 
\pr{BA} = \pr{A}
\end{align}
From \eqref{eq:ncert/12/13/6/1/1}, 
\begin{align}
\pr{B \mid A}= 1
\end{align}
\item $A \cap B = \phi$. This implies
\begin{align}
\pr{BA} = 0
\end{align}
From \eqref{eq:ncert/12/13/6/1/1},
\begin{align}
\pr{B \mid A}= 0
\end{align}
\end{enumerate}
