\begin{enumerate}
\item 
			From 
			\eqref{moment:def}
			and
			\eqref{moment:def-gen}
\begin{align}
E\brak{X}&=\sum_{k=0}^{4}kp_X\brak{k}
=2.05
\end{align}
and
\begin{align}
E\brak{X^2}&=\sum_{k=0}^{4}k^2p_X\brak{k}
=5.65
\end{align}
Substituting from the above in \eqref{var:moment},
\begin{align}
\text{var}\brak{X}&=5.65-\brak{2.05}^2
=1.4475
\end{align}
\item 
		From	\eqref{var:homo-l},
\begin{align}
	\text{var}\brak{\frac{X}{2}}=\frac{\text{var}\brak{X}}{4}
=0.361875
\end{align}
\end{enumerate}
