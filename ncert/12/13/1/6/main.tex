		If $X_i \sim \ber{\frac{1}{2}}, i = 1, 2, 3$ with 0 denoting head, 
\begin{enumerate}
\item  The events $E, F$ can be described as
\begin{align}
E &: X_3 = 0\\
F &: X_1 + X_2 = 0 
\end{align}
Therefore, 
\begin{align}
	\pr{E|F} &= \frac{\pr{EF}}{\pr{F}} = \frac{\pr{X_3=0, X_1+X_2 = 0}}{\pr{X_1+X_2 = 0}} 
	\\
	&= \frac{\pr{X_1+X_2+X_3 = 0}}{\pr{X_1+X_2 = 0}}= \frac{\comb{3}{0}\brak{\frac{1}{2}}^{3}}{\comb{2}{0}\brak{\frac{1}{2}}^{2}}=\frac{1}{2}
\end{align} 
\item  The desired probability can be expressed as
\begin{align}
\frac{\pr{EF}}{1- \pr{F^{\prime}}}
&= \frac{\pr{X = 1}}{1 - \pr{X = 0}}  \label{eq:ncert/12/13/1/6/5}\\
&= \frac{\frac{3}{8}}{1-\frac{1}{8}}
= \frac{3}{7}
\end{align}
where
\begin{align}
X=X_1+X_2+X_3. 
\end{align}
\item  In this case,  
the desired probability is 
\begin{align}
\frac{\pr{X \le 2, X \ge 1}}{\pr{X \ge 1}}
=\frac{\pr{1 \le X \le 2}}{\pr{X \ge 1}}
= \frac{6}{7}
\end{align}

\end{enumerate}
