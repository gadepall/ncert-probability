Given:
\begin{align}
    \pr{AB} &= 0.3\\
    \pr{A + B} &= 0.6\\
    &=  \pr{A} + \pr{B} - \pr{AB}\\
    \implies 0.6 &= \pr{A} + \pr{B} - 0.3\\
    \implies 0.9 &= \pr{A} + \pr{B}
\end{align}
But
\begin{align}
    \pr{A^\prime} &= 1 - \pr{A}\\
    \pr{B^\prime} &= 1 - \pr{B}\\
    \therefore \pr{A^\prime} + \pr{B^\prime} &= 2 - \brak{\pr{A} + \pr{B}}\\
    &= 2 - 0.9
    = 1.1
\end{align}
