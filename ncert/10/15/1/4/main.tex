\solution

Probability of an event $E$ , written as \pr{E}\:

\begin{align}
\label{eq:ncert/10/15/1/4/def}
\pr{E}=\frac{\text{Number of outcomes favourable to $E$}}{\text{Total Number of possible outcomes in sample space}}
\end{align}
\fi
From the axioms of probability,
\begin{align}
   \label{eq:ncert/10/15/1/4/prob}
   0 \le \pr{E}\le 1 
\end{align}
\begin{enumerate}
\item \pr{E} = $\frac{2}{3}$
\begin{align}
    \because 0 \le \frac{2}{3} \le 1
\end{align}
from \eqref{eq:ncert/10/15/1/4/prob},
 it can be probability of an event.
\item \pr{E} = $-1.5$
\begin{align}
\because  -1.5 < 0 
\end{align}
from \eqref{eq:ncert/10/15/1/4/prob},
it cannot be a probability of any event.
\item 
\begin{align}
	\pr{E} &= 
\frac{15}{100}
\\
	\because 0 &\le \frac{15}{100} \le 1,
\end{align}
from \eqref{eq:ncert/10/15/1/4/prob},
it can be probability of an event.
%
\item \pr{E} = $0.7$ 
\begin{align}
\because 0 \le 0.7 \le 1 
\end{align}
from \eqref{eq:ncert/10/15/1/4/prob},
it can be a probability of an event.
\end{enumerate}
