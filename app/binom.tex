\item The Binomial distribution is defined as 
\begin{align}
\label{eq:binom-ber}
X &= X_1+X_2+\dots+X_n,
\end{align}
Where $X_i$ are i.i.d bernoulli.
\item For $X_i\sim \text{Ber}\brak{p}$,
\begin{align}
\label{eq:mgf-ber}
M_{X_i}(z)= q+pz^{-1}, \quad q = 1-p
\end{align}
upon substituting from 
\eqref{eq:ber-pmf}
in
\eqref{eq:dice_xz}.
\item For $X\sim \text{Binom}\brak{n,p}$,
\begin{align}
\label{eq:mgf-binom}
M_X(z)= \brak{q+pz^{-1}}^n
\end{align}
using
\eqref{eq:binom-ber}
and
\eqref{eq:dice_xzprod_def}.
\item Expanding
\eqref{eq:mgf-binom},
\begin{align}
	M_X(z)&= \sum_{k=0}^{n}\brak{1-p}^{n-k}p^kz^{-k}
	\\
	\implies p_X(k)
	      &=
	\begin{cases}
	\brak{1-p}^{n-k}p^k & 0 \le k \le n
	\\
	0 & \text{otherwise}
	\end{cases}
\end{align}
upon comparing with \eqref{eq:dice_xz}.
